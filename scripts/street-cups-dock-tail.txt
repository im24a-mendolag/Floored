        <div className="min-h-10 flex w-full items-center justify-center px-2 shrink-0">
          <p className="text-center text-xs text-zinc-500">
            {isBetting
              ? 'Memorise the crown, follow the shuffle, then pick a cup.'
              : isRevealing
                ? 'Memorise the crown…'
                : isShuffling
                  ? 'Follow the cups…'
                  : isPicking
                    ? 'Tap a cup on the board.'
                    : '\u00A0'}
          </p>
        </div>
      </GameFieldWithHistory>

      <motion.div className={GAME_CONTROL_DOCK_M}>
        <motion.div className={GAME_DOCK_INNER}>
          <GameDockChipRow
            visible={isBetting || showQuoteUntilNext}
            bankroll={bankroll}
            currentBet={currentBet}
            onAddChip={addChip}
            quoteIdx={quoteIdx}
            showQuote={showQuoteUntilNext}
          />

          <motion.div className="h-10 flex items-center justify-center">
            {isBetting && <GameDockBetRow currentBet={currentBet} onClear={() => setCurrentBet(0)} />}
            {isPicking && potentialWinnings > 0 && (
              <p className="text-sm text-zinc-400">
                Potential total winnings:{' '}
                <span className="font-semibold text-emerald-400">{formatChips(potentialWinnings)}</span>
              </p>
            )}
            {(isRevealing || isShuffling) && (
              <p className="text-sm text-zinc-500">
                {isRevealing ? 'Memorise the crown…' : 'Follow the cups…'}
              </p>
            )}
            {isSettled && pendingResult && showNext && (
              <GameDockSettledRow
                outcomeLabel={pendingResult.outcomeLabel}
                label={pendingResult.label}
                tone={pendingResult.tone}
              />
            )}
            {!isBetting && !isPicking && !(isSettled && pendingResult && showNext) && !(isRevealing || isShuffling) && (
              <p className="text-sm invisible select-none">{'\u00A0'}</p>
            )}
          </motion.div>

          <motion.div className="flex justify-center">
            <button
              type="button"
              onClick={isSettled && showNext ? handleNext : handleStart}
              disabled={actionDisabled}
              className="min-w-[10.5rem] px-7 py-2 bg-white hover:bg-zinc-100 disabled:bg-zinc-800 disabled:text-zinc-600 text-zinc-900 font-bold rounded-lg transition-colors text-base shadow-lg"
            >
              {actionLabel}
            </button>
          </motion.div>

          {minBet > 1 && isBetting && (
            <p className="text-center text-zinc-600 text-sm">Min bet: {formatChips(minBet)}</p>
          )}
        </motion.div>
      </motion.div>
    </motion.div>
  )
}
